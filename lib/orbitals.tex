\newcommand{\addOrbital}[1]{
    \pgfkeys{/mohelper/.cd,
        energy/.store in=\tempEnergy,
        energy=0,
        sym/.store in=\tempSym,
        sym=a1,
        % deg/.store in=\tempDeg,
        % deg=1,
        config/.store in=\tempConfig,
        config={empty},
        labels/.store in=\tempLabels,
        labels={},
        side/.store in=\tempSide,
        side=left,
        #1
    } % Verarbeite die User-Eingaben

    \xdef\mergedstring{}
    \foreach \spin [count=\ione] in \tempConfig {
        \edef\labelorbital{}
        \foreach \ee [count=\itwo] in \tempLabels {
            \ifdefequal{\ione}{\itwo}{
                \xdef\labelorbital{\ee}
            }{}
        }
        \ifdefempty{\mergedstring}{
            % Erster Eintrag: Kein Komma davor
            \xdef\mergedstring{\labelorbital/\spin}
        }{
            % Alle weiteren Einträge: Mit Komma trennen
            \xdef\mergedstring{\mergedstring,\labelorbital/\spin}
        }
    }
    \StrCount{\tempConfig}{,}[\arrcount]%
    \pgfmathsetmacro{\tempDeg}{int(\arrcount + 1)}%

    \degenerate{\tempEnergy}{\mergedstring}{\tempSide}

    \degeneratelabel{\tempEnergy}{\tempDeg}{\tempSide}{top}{data=\tempSym/\tempEnergy/\tempDeg}
    % \degeneratelabel{\tempEnergy}{1}{left}{bottom}{config=\tempSide}
    \degeneratelabel{\tempEnergy}{\tempDeg}{\tempSide}{right}{\mergedstring}

    \foreach \group in \tempSym {

        \listcsxadd{orbit@list\tempSide}{\group/\tempEnergy/\tempDeg}
    }




    % \expandafter\typeout\expandafter{\csname orbit@listleft\endcsname}
}


\newcommand{\cslisttoarray}[2]{%
    % #1 = Name der Quelle (z.B. orbit@list@left)
    % #2 = Name des Ziel-Makros (z.B. \myArray)
    \xdef#2{}% Ziel leeren
    \def\do##1{%
        \ifdefempty{#2}
        {\xdef#2{##1}}% Erstes Element
        {\xdef#2{#2,##1}}% Weitere Elemente mit Komma
    }%
    \dolistcsloop{#1}%
}

\newcommand{\autoconnect}{

    \cslisttoarray{orbit@listleft}{\arrayLeft}
    \cslisttoarray{orbit@listmiddle}{\arrayMiddle}
    \cslisttoarray{orbit@listright}{\arrayRight}

    \foreach \symmetry/\energy/\deg in \arrayLeft{

        \foreach \symmetryy/\energyy/\degy in \arrayMiddle{

            \ifdefequal{\symmetry}{\symmetryy}{
                \connectorbital{\energy}{\deg}{\energyy}{\degy}{left}
            }{}

        }
    }
    \foreach \symmetry/\energy/\deg in \arrayMiddle{

        \foreach \symmetryy/\energyy/\degy in \arrayRight{

            \ifdefequal{\symmetry}{\symmetryy}{
                \connectorbital{\energy}{\deg}{\energyy}{\degy}{right}
            }{}

        }
    }


}

\makeatother
